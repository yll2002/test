void time()
{
	SYSTEMTIME st = {0};
    GetLocalTime(&st);
    printf("%d-%02d-%02d %02d:%02d:%02d", 
    st.wYear,

    st.wDay,
    st.wHour,
    st.wMinute,
    st.wSecond);
}

void todo(string a)

	}

	}

		int x;

	{
		cout<<"开发者QQ：2764863869。"<<endl;
		cout<<"开发者：叶礼亮。"<<endl;
	}
	if (a=="time")
	{
		time();

	}

		char b[100];
